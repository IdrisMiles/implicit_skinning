1012 inline int _ConvertSMVer2Cores(int major, int minor)
1013 {
1014  // Defines for GPU Architecture types (using the SM version to determine the # of cores per SM
1015  typedef struct
1016  {
1017  int SM; // 0xMm (hexidecimal notation), M = SM Major version, and m = SM minor version
1018  int Cores;
1019  } sSMtoCores;
1020 
1021  sSMtoCores nGpuArchCoresPerSM[] =
1022  {
1023  { 0x20, 32 }, // Fermi Generation (SM 2.0) GF100 class
1024  { 0x21, 48 }, // Fermi Generation (SM 2.1) GF10x class
1025  { 0x30, 192}, // Kepler Generation (SM 3.0) GK10x class
1026  { 0x32, 192}, // Kepler Generation (SM 3.2) GK10x class
1027  { 0x35, 192}, // Kepler Generation (SM 3.5) GK11x class
1028  { 0x37, 192}, // Kepler Generation (SM 3.7) GK21x class
1029  { 0x50, 128}, // Maxwell Generation (SM 5.0) GM10x class
1030  { 0x52, 128}, // Maxwell Generation (SM 5.2) GM20x class
1031  { -1, -1 }
1032  };
1033 
1034  int index = 0;
1035 
1036  while (nGpuArchCoresPerSM[index].SM != -1)
1037  {
1038  if (nGpuArchCoresPerSM[index].SM == ((major << 4) + minor))
1039  {
1040  return nGpuArchCoresPerSM[index].Cores;
1041  }
1042 
1043  index++;
1044  }
1045 
1046  // If we don't find the values, we default use the previous one to run properly
1047  printf("MapSMtoCores for SM %d.%d is undefined. Default to use %d Cores/SM\n", major, minor, nGpuArchCoresPerSM[index-1].Cores);
1048  return nGpuArchCoresPerSM[index-1].Cores;
1049 }

1100 // This function returns the best GPU (with maximum GFLOPS)
1101 inline int gpuGetMaxGflopsDeviceId()
1102 {
1103  int current_device = 0, sm_per_multiproc = 0;
1104  int max_perf_device = 0;
1105  int device_count = 0, best_SM_arch = 0;
1106  int devices_prohibited = 0;
1107 
1108  unsigned long long max_compute_perf = 0;
1109  cudaDeviceProp deviceProp;
1110  cudaGetDeviceCount(&device_count);
1111 
1112  checkCudaErrors(cudaGetDeviceCount(&device_count));
1113 
1114  if (device_count == 0)
1115  {
1116  fprintf(stderr, "gpuGetMaxGflopsDeviceId() CUDA error: no devices supporting CUDA.\n");
1117  exit(EXIT_FAILURE);
1118  }
1119 
1120  // Find the best major SM Architecture GPU device
1121  while (current_device < device_count)
1122  {
1123  cudaGetDeviceProperties(&deviceProp, current_device);
1124 
1125  // If this GPU is not running on Compute Mode prohibited, then we can add it to the list
1126  if (deviceProp.computeMode != cudaComputeModeProhibited)
1127  {
1128  if (deviceProp.major > 0 && deviceProp.major < 9999)
1129  {
1130  best_SM_arch = MAX(best_SM_arch, deviceProp.major);
1131  }
1132  }
1133  else
1134  {
1135  devices_prohibited++;
1136  }
1137 
1138  current_device++;
1139  }
1140 
1141  if (devices_prohibited == device_count)
1142  {
1143  fprintf(stderr, "gpuGetMaxGflopsDeviceId() CUDA error: all devices have compute mode prohibited.\n");
1144  exit(EXIT_FAILURE);
1145  }
1146 
1147  // Find the best CUDA capable GPU device
1148  current_device = 0;
1149 
1150  while (current_device < device_count)
1151  {
1152  cudaGetDeviceProperties(&deviceProp, current_device);
1153 
1154  // If this GPU is not running on Compute Mode prohibited, then we can add it to the list
1155  if (deviceProp.computeMode != cudaComputeModeProhibited)
1156  {
1157  if (deviceProp.major == 9999 && deviceProp.minor == 9999)
1158  {
1159  sm_per_multiproc = 1;
1160  }
1161  else
1162  {
1163  sm_per_multiproc = _ConvertSMVer2Cores(deviceProp.major, deviceProp.minor);
1164  }
1165 
1166  unsigned long long compute_perf = (unsigned long long) deviceProp.multiProcessorCount * sm_per_multiproc * deviceProp.clockRate;
1167 
1168  if (compute_perf > max_compute_perf)
1169  {
1170  // If we find GPU with SM major > 2, search only these
1171  if (best_SM_arch > 2)
1172  {
1173  // If our device==dest_SM_arch, choose this, or else pass
1174  if (deviceProp.major == best_SM_arch)
1175  {
1176  max_compute_perf = compute_perf;
1177  max_perf_device = current_device;
1178  }
1179  }
1180  else
1181  {
1182  max_compute_perf = compute_perf;
1183  max_perf_device = current_device;
1184  }
1185  }
1186  }
1187 
1188  ++current_device;
1189  }
1190 
1191  return max_perf_device;
1192 }
